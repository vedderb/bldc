90 * they should be quite fast. They are (more or less) equivalent with the following:

92 * float frexp_slow(float f, int *e) {
93 * if (f == 0.0) {
94 * *e = 0;
95 * return 0.0;
96 * }

99 * float res = f / powf(2.0, (float)*e);
100 *
101 * if (res >= 1.0) {
102 * res -= 0.5;
103 * *e += 1;
104 * }
105 *
106 * if (res <= -1.0) {
107 * res += 0.5;
108 * *e += 1;
109 * }
110 *
111 * return res;

114 * float ldexp_slow(float f, int e) {
